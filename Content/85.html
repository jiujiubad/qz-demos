CREATE TABLE people (  
    id integer, 
    first_name  varchar(50)  NOT FULL, 
    last_name  varchar(50) NOT FULL, 
    gender varchar(8)  DEFAULT “M" ,
    birthday  DATE
  
) 
  • 请写出 SQL 捞出所有资料
  • SELECT * FROM people
  • 请写出 SQL 捞出所有男生
  • SELECT* FROM people WHERE  gender = “M"
  • 请写出 SQL 插入一笔资料
  • INSERT INTO people(first_name,last_name,gender,birthday) VALUES("echo","zhou","F",20140404)
  • 请写出 SQL 更新 id 为 3 的资料，将姓名修改成你的名字
  • UPDATE people SET first_name = “echo”,last_name= “zhou”  WHERE id = 3
  • 请写出 SQL 删除 id 为 4 的资料
  • DELETE FROM people WHERE id =4
