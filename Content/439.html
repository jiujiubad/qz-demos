def sum(x,y)  # x代表买的本数，y代表买的集数
  sum = 0
  case x
  when   1
    sum = 100 * x
  when   2
    sum = y * 100 - x * 100 * 0.05
  when   3
    sum = y * 100 - x * 100 * 0.1
  when   4
    sum = y * 100 - x * 100 * 0.2
  when   5
    sum = y * 100 - x * 100 * 0.25
  end
end
