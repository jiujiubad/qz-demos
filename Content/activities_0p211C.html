class SellBooks
  def initialize(*books)
    @size = books.size
    @books = {}
    books.each do |b|
      if @books.has_key? b
        @books[b] += 1
      else
        @books[b] = 0
      end
    end
  end

  def discount
    puts @books.size
    case @books.size
      when 1
        1.0
      when 2
        0.95
      when 3
        0.9
      when 4
        0.8
      else
        0.75
    end
  end

  def total_fee
    @size * 100 * discount
  end
end
